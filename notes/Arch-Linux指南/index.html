# 基础系统安装

# 确认是否为 UEFI 模式

若输出了一堆东西，即 efi 变量，则说明已在 UEFI 模式。否则请确认你的启动方式是否为 UEFI。

# 启动参数修改

  • 添加 modprobe.blacklist=nouveau 以禁用开源驱动

# 使用 iwd 联网

  • iwctl
  • device list
  • station device scan
  • station wlan0 connect "Redmi_xxx"

# 更新系统时间

  • timedatectl set-ntp true
  • timedatectl status

# 分区

  • lsblk
  • gdisk /dev/nvme0n1
  • ef00、8400
  • mkfs.fat -F32 /dev/nvme0n1p1
  • mkswap /dev/nvme0n1p2
  • swapon /dev/nvme0n1p2
  • mkfs.xfs /dev/nvme0n1p3
  • mkfs.xfs /dev/nvme0n1p4
  • mount /dev/nvme0n1p3 /mnt
  • mkdir /mnt/boot
  • mkdir /mnt/home
  • mount /dev/nvme0n1p1 /mnt/boot
  • mount /dev/nvme0n1p4 /mnt/home

# 选择镜像

  • vim /etc/pacman.d/mirrorlist
  • Server = https://mirrors.bfsu.edu.cn/archlinux/$repo/os/$arch

# 安装必须软件包

pacstrap /mnt bash-completion iwd dhcpcd base base-devel linux linux-firmware linux-headers man man-db man-pages texinfo vim xfsprogs nvidia nvidia-utils nvidia-settings opencl-nvidia

# 生成 Fstab

genfstab -U /mnt >> /mnt/etc/fstab

vim /mnt/etc/fstab

# Chroot

arch-chroot /mnt

# 本地化

ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime

hwclock --systohc

vim /etc/locale.genq 取消注释

locale-gen

vim /etc/locale.conf LANG=en_US.UTF-8

# 网络配置

vim /etc/hostname

AORUS

vim /etc/hosts

127.0.0.1	localhost
::1		localhost
127.0.1.1	AORUS.localdomain	AORUS

# Initramfs

mkinitcpio -P

# 密码

passwd

# 安装引导程序

pacman -Sy grub efibootmgr os-prober xfsprogs ntfs-3g

mkdir MS

mount /dev/nvme1n1p1 MS

grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=Arch --recheck

grub-mkconfig -o /boot/grub/grub.cfg

exit

umount -R /mnt

reboot

# 桌面环境安装

# 联网

systemctl start iwd

dhcpcd

iwctl

station wlan0 connect "Redmi_xxx"

# 新建用户并授权

useradd -m -G wheel limbo

EDITOR=vim visudo

%wheel ALL=(ALL) ALL

exit

# 安装桌面

sudo pacman -S xorg gdm gnome gnome-tweaks

# 自启动设置

sudo systemctl preset-all

sudo systemctl enable gdm

sudo systemctl enable NetworkManager

sudo systemctl enable bluetooth

# 配置 CN 源

vim /etc/pacman .conf

[archlinuxcn]

Server = https://mirrors.bfsu.edu.cn/archlinuxcn/$arch

# 安装软件

sudo pacman -S zsh alacritty git wget google-chrome firefox neofetch gcc gdb clang llvm nodejs yarn

# 安装 Fcitx5

sudo pacman -S fcitx5-im fcitx5-chinese-addons

另外 CN 源有词库可用：

sudo pacman -S fcitx5-pinyin-{zhwiki,moegirl}

主题参考：

https://github.com/hosxy/Fcitx5-Material-Color

多看看 https://www.gnome-look.org/browse/cat/

xcursor-breeze

扩展：

appindicator and kstatusnotifieritem support

arcmenu

caffeine

dash to dock

openweather

transparent top bar

user themes

sudo pacman -S alacritty

sudo pacman -S neovim nodejs yarn python-pynvim xsel

# 修 gdm 和 nvidia 冲突 bug！

  • 让 nvidia 启动写进 kernel modules，提前启动

sudo nvim /etc/mkinitcpio.conf

MODULES=(nvidia nvidia_modeset nvidia_uvm nvidia_drm)

HOOKS=(base udev autodetect modconf block filesystems keyboard fsck)

sudo nvim /etc/default/grub

内核参数加 nvidia-drm.modeset=1

sudo mkinitcpio -P

sudo grub-mkconfig -o /boot/grub/grub.cfg
